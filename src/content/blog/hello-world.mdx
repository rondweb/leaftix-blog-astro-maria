### Python

```python
# Digital Twin — Variance Simulation
import numpy as np
from dataclasses import dataclass

@dataclass
class AgriculturalTwin:
    temperature: float
    humidity: float
    soil_moisture: float
    biomass_yield: float

    def simulate_variance(self) -> dict:
        """Run Monte Carlo simulation for yield prediction."""
        simulations = 10_000
        results = np.random.normal(
            loc=self.biomass_yield,
            scale=self.biomass_yield * 0.02,  # < 2% variance
            size=simulations
        )
        return {
            'mean': np.mean(results),
            'std': np.std(results),
            'variance_pct': (np.std(results) / np.mean(results)) * 100
        }

twin = AgriculturalTwin(24.5, 65.0, 0.42, 340)
print(twin.simulate_variance())
# Output: variance_pct ≈ 1.97%
```
